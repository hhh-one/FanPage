CREATE TABLE "MEMBERS" (
	"NAME"	VARCHAR2(20)		NULL,
	"ID"	VARCHAR2(20)		NOT NULL,
	"PW"	VARCHAR2(30)		NOT NULL,
	"GENDER"	CHAR(1)		    NULL,
	"EMAIL"	VARCHAR2(50)		NULL
);

CREATE TABLE "BOARDS" (
	"BNO"	NUMBER(10)		    NOT NULL,
	"NAME"	VARCHAR2(20)		NULL,
	"TITLE"	VARCHAR2(30)		NOT NULL,
	"CONTENT"	VARCHAR2(1000)	NULL,
	"REGDATE"	DATE		    DEFAULT SYSDATE,
	"HIT"	NUMBER(10)		    DEFAULT 0
);

CREATE TABLE "TALKTALK" (
	"TNO"	NUMBER(10)		    NOT NULL,
	"NAME"	VARCHAR2(20)		NULL,
	"CONTENT"	VARCHAR2(200)	NULL,
	"REGDATE"	DATE		    DEFAULT SYSDATE
);

CREATE TABLE "COMMENTS" (
	"CNO"	NUMBER(10)		    NOT NULL,
	"NAME"	VARCHAR2(20)		NOT NULL,
	"BNO"	NUMBER(10)		    NOT NULL,
	"CONTENT"	VARCHAR2(1000)	NULL,
	"REGDATE"	DATE		    DEFAULT SYSDATE
);

ALTER TABLE "MEMBERS" ADD CONSTRAINT "PK_MEMBERS" PRIMARY KEY (
	"NAME"
);

ALTER TABLE "BOARDS" ADD CONSTRAINT "PK_BOARDS" PRIMARY KEY (
	"BNO"
);

ALTER TABLE "TALKTALK" ADD CONSTRAINT "PK_TALKTALK" PRIMARY KEY (
	"TNO"
);

ALTER TABLE "COMMENTS" ADD CONSTRAINT "PK_COMMENTS" PRIMARY KEY (
	"CNO"
);

ALTER TABLE BOARDS ADD CONSTRAINT "FK_BOARDS_NAME" FOREIGN KEY (
    "NAME"
) REFERENCES MEMBERS (
    "NAME"
);

ALTER TABLE TALKTALK ADD CONSTRAINT "FK_TALKTALK_NAME" FOREIGN KEY (
    "NAME"
) REFERENCES MEMBERS (
    "NAME"
);

ALTER TABLE COMMENTS ADD CONSTRAINT "FK_COMMENTS_NAME" FOREIGN KEY (
    "NAME"
) REFERENCES MEMBERS (
    "NAME"
);

ALTER TABLE COMMENTS ADD CONSTRAINT "FK_COMMENTS_BNO" FOREIGN KEY (
    "BNO"
) REFERENCES BOARDS (
    "BNO"
);

ALTER TABLE MEMBERS ADD CONSTRAINT MEMBERS_CHECK CHECK(GENDER IN('M', 'F'));

CREATE SEQUENCE BOARDS_SEQ NOCACHE;

CREATE SEQUENCE TALKTALK_SEQ NOCACHE;

CREATE SEQUENCE COMMENTS_SEQ NOCACHE;
